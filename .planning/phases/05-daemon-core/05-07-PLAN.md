```typescript
function createHttpDefinition(url: string, bearerToken: string): unknown {
  const httpCtor = (vscode as unknown as { McpHttpServerDefinition?: McpCtor }).McpHttpServerDefinition;
  if (!httpCtor) return null;
  try {
    return new httpCtor({ url, headers: { Authorization: `Bearer ${bearerToken}` } });
  } catch {
    return new httpCtor(url, { Authorization: `Bearer ${bearerToken}` });
  }
}
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-07-01 | Spoofing | McpHttpServerDefinition duck-type | mitigate | Falls back to stdio if constructor throws; duck-type check prevents undefined constructor call |
| T-05-07-02 | Information Disclosure | bearer token in HTTP MCP definition | mitigate | Token passed in headers object to VS Code API; not logged; loopback URL only |
  T-05-07-03 | Denial of Service | useDaemon=true but daemon never starts | accept | probeHealth returns false -> falls back to stdio definition; user still gets MCP functionality |
