```typescript
export interface DaemonStatus {
  running: boolean;
  healthy: boolean;
  pid: number | null;
  port: number | null;
  port_lsp: number | null;
  bearerToken: string | null;
  tunnelUrl: string | null;
  uptime: number | null; // ms since startedAt
}

export interface DaemonConnectionInfo {
  pid: number;
  uuid: string;
  port: number;
  url: string;           // http://127.0.0.1:{port}
  bearerToken: string;
  version: string;
  startedAt: string;
}

export class DaemonManager implements vscode.Disposable {
  private readonly _onDidChange = new vscode.EventEmitter<DaemonStatus>();
  public readonly onDidChange = this._onDidChange.event;
  private _disposed = false;

  constructor(
    private readonly configDir: string,
    private readonly extensionPath: string,
  ) {}

  async ensureDaemon(options?: { timeoutMs?: number }): Promise<DaemonConnectionInfo>
  async stopDaemon(): Promise<void>
  async restartDaemon(): Promise<DaemonConnectionInfo>
  async getDaemonStatus(): Promise<DaemonStatus>
  async probeHealth(): Promise<boolean>
  buildDaemonEnv(credEnv?: Record<string, string>): Record<string, string>

  dispose(): void {
    this._disposed = true;
    this._onDidChange.dispose();
  }
}
```

    getDaemonStatus(): reads daemon.lock via fs.readFile; parses JSON; calls probeHealth; returns DaemonStatus. On any error (file not found, parse error): returns running:false/healthy:false/all-null status.

    probeHealth(): reads current status from getDaemonStatus; if running:false or no port/bearerToken, returns false. Uses AbortController with 2s timeout; fetches GET http://127.0.0.1:{port}/daemon/health with Authorization: Bearer {token}; returns response.ok; catches all errors and returns false.

    ensureDaemon(): checks _disposed; sets deadline = Date.now() + (options.timeoutMs ?? 15_000); polls getDaemonStatus every 200ms; calls _spawnDetached once when not running; returns DaemonConnectionInfo on success; throws on deadline exceeded.

    _spawnDetached(): private method. Calls getBundledServerPath to get the path to dist/mcp/index.mjs. Spawns: `spawn(process.execPath, [serverPath, 'daemon', 'start'], { detached: true, stdio: 'ignore', env: { ...process.env, ...this.buildDaemonEnv() } })`. Calls child.unref().

    stopDaemon(): reads getDaemonStatus; if not running, returns. Fetches POST http://127.0.0.1:{port}/daemon/shutdown with bearer token and 3s timeout. Ignores response (daemon terminates itself).

    restartDaemon(): calls stopDaemon(); waits 500ms; calls ensureDaemon(); returns result.

    buildDaemonEnv(): as shown in interfaces block.

    dispose(): `this._disposed = true; this._onDidChange.dispose();`

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Extension (VS Code host) → daemon process (child) | buildDaemonEnv passes auth env vars; includes AIRTABLE_HEADLESS_ONLY to prevent interactive browser |
| Extension probeHealth fetch → daemon HTTP (loopback) | Bearer token in Authorization header; loopback only |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-06-01 | Information Disclosure | buildDaemonEnv includes auth credentials | mitigate | Credentials (AIRTABLE_EMAIL etc.) only injected when passed via credEnv parameter from AuthManager; not injected by default |
| T-05-06-02 | Information Disclosure | bearer token in probeHealth fetch | mitigate | Fetch is to loopback (127.0.0.1) only; AbortController ensures 2s timeout prevents indefinite exposure |
  T-05-06-03 | Information Disclosure | bearer token in getDaemonStatus | mitigate | getDaemonStatus reads token from lockfile but must NOT log it; only port and pid are safe to log |
  T-05-06-04 | Denial of Service | ensureDaemon 15s timeout | accept | 15s is a fixed upper bound; throws a clear error to the caller on timeout |
</threat_model>
