**Step 2: Add enableTunnel action**

```typescript
enableTunnel: async (provider: TunnelProviderId, authtoken?: string, domain?: string) => {
  const id = nanoid();
  return new Promise<{ ok: boolean; error?: string }>((resolve) => {
    // Register result handler
    pendingActions.set(id, resolve);
    sendMessage({ type: 'tunnel:enable', id, provider, authtoken, domain });
    // Timeout after 30 seconds (tunnel start can take time)
    setTimeout(() => {
      if (pendingActions.has(id)) {
        pendingActions.delete(id);
        resolve({ ok: false, error: 'Timeout starting tunnel' });
      }
    }, 30_000);
  });
},
```

**Step 3: Add disableTunnel action**

```typescript
disableTunnel: async () => {
  const id = nanoid();
  return new Promise<{ ok: boolean; error?: string }>((resolve) => {
    pendingActions.set(id, resolve);
    sendMessage({ type: 'tunnel:disable', id });
    setTimeout(() => {
      if (pendingActions.has(id)) {
        pendingActions.delete(id);
        resolve({ ok: false, error: 'Timeout disabling tunnel' });
      }
    }, 10_000);
  });
},
```

**Step 4: Add setNgrokAuthtoken action**

```typescript
setNgrokAuthtoken: async (authtoken: string) => {
  const id = nanoid();
  return new Promise<{ ok: boolean; error?: string }>((resolve) => {
    pendingActions.set(id, resolve);

      if (pendingActions.has(id)) {
        pendingActions.delete(id);
        resolve({ ok: false, error: 'Timeout storing authtoken' });

    }, 5_000);
  });
},
```

```tsx
const handleEnableTunnel = async () => {
  setIsTunnelPending(true);
  try {
    // If ngrok and new authtoken entered, store it first
    if (selectedProvider === 'ngrok' && ngrokAuthtokenInput) {
      await setNgrokAuthtoken(ngrokAuthtokenInput);
    }
    await enableTunnel(selectedProvider, undefined, ngrokDomainInput || undefined);
  } finally {
    setIsTunnelPending(false);
  }
};

const handleDisableTunnel = async () => {
  setIsTunnelPending(true);
  try {
    await disableTunnel();
  } finally {
    setIsTunnelPending(false);
  }
};

const handleCopyUrl = () => {
  if (tunnel?.url) {
    navigator.clipboard.writeText(tunnel.url).catch(() => undefined);
    setCopiedUrl(true);
    setTimeout(() => setCopiedUrl(false), 1500);
  }
};
```
