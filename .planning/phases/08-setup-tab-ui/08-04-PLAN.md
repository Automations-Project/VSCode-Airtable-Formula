Replace the entire stub (including the comment) with the full implementation:

```typescript
export function getMcpSnippet(ide: string, variant: 'http' | 'stdio', port: number | string): string {
  if (variant === 'stdio') {
    return `"airtable": {
  "command": "npx",
  "args": ["-y", "airtable-user-mcp"]
}`;
  }

  // HTTP variant — each IDE has a slightly different key name (Pitfall 2 in RESEARCH.md)
  // Claude Code + Claude Desktop: type: "http" + url
  // Cursor + Cline: url only (no type key)
  // Windsurf: serverUrl (not url)
  if (ide === 'windsurf') {
    return `"airtable": {
  "serverUrl": "http://127.0.0.1:${port}/mcp",
  "headers": {
    "Authorization": "Bearer {{BEARER_TOKEN}}"
  }
}`;
  }

  if (ide === 'cursor' || ide === 'cline') {
    return `"airtable": {
  "url": "http://127.0.0.1:${port}/mcp",
  "headers": {
    "Authorization": "Bearer {{BEARER_TOKEN}}"
  }
}`;
  }

  // claude-code, claude-desktop (and any future IDE) — use type: "http" + url
  return `"airtable": {
  "type": "http",
  "url": "http://127.0.0.1:${port}/mcp",
  "headers": {
    "Authorization": "Bearer {{BEARER_TOKEN}}"
  }
}`;
}
```

**Security note (D-07, T-08-01):** The literal string `{{BEARER_TOKEN}}` is hardcoded — no variable interpolation from state or props. The `bearerToken` field was stripped from DaemonStatusInfo in Plan 02. This function cannot accidentally leak a real token.
